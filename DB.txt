DROP TABLE [IF EXISTS] posts;

CREATE TABLE posts(
	id serial PRIMARY KEY,
	title TEXT,
	description TEXT,
	tags TEXT,
	content TEXT NOT NULL,
	comments INT NOT NULL,
	views INTEGER NOT NULL,
	created TIMESTAMP,
	updated TIMESTAMP,
	image TEXT,
	url TEXT,
	category TEXT,
	status VARCHAR(10) NOT NULL
) INHERITS posts;
